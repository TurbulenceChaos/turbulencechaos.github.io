Plot style:

GraphicsGrid[{{
    Labeled[plot1, "(a) plot 1", Bottom, Spacings -> 2],
    Labeled[plot2, "(b) plot 2", Bottom, Spacings -> 2]
  }}]
FrameLabel -> {Row["x\n", Style["title", FontSize -> 12]], None}
plotStyle = Sequence[PlotTheme -> {"Scientific", "Monochrome"},
   BaseStyle -> {FontSize -> 10, FontFamily -> "Times"},
   LabelStyle -> {FontSize -> 10, FontFamily -> "Times"}
   ];
plot=Plot[Sin[x], {x, 0, 1}, Evaluate[plotStyle]]

Plot export:

PlotExport[name_, plot_, width_] := 
  Export[name, plot, ImageResolution -> 600, 
   RasterSize -> Round[width/2.54*600]];
