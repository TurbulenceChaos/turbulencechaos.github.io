Wolfram Mathematica

Initialize the notebook:

If[Length[Names["`*"]] > 0, Remove["`*"]];
SetDirectory[NotebookDirectory[]];

Plot style:

plotStyle[width_ : 8] := 
   Sequence[PlotTheme -> {"Scientific", "Monochrome"},
   BaseStyle -> {FontSize -> 10, FontFamily -> "Times"},
   LabelStyle -> {FontSize -> 10, FontFamily -> "Times"},
   ImageSize -> width/2.54*100
   ];

plot=Plot[Sin[x], {x, 0, 1}, Evaluate[plotStyle[]]]

Plot export:

PlotExport[file_, plot_, width_ : 8] := Module[{dir},
   dir = DirectoryName[file];
   dir = If[dir === "", NotebookDirectory[], dir];
   If[! DirectoryQ[dir], 
    CreateDirectory[dir, CreateIntermediateDirectories -> True]];
   Export[file, plot, ImageResolution -> 600, 
    RasterSize -> Round[width/2.54*600]]
   ];
  
PlotExport["plot.png", plot];

OpenFOAM

Installation:

#!/bin/bash

cd $HOME/openfoam/

sudo rm -rf OpenFOAM-v2512
sudo tar -xzf OpenFOAM-v2512.tgz -C .
sudo rm -rf ThirdParty-v2512
sudo tar -xzf  ThirdParty-v2512.tar.gz -C .

sudo chmod -R 777 OpenFOAM-v2512
sudo chmod -R 777 ThirdParty-v2512

cd OpenFOAM-v2512

source ./etc/bashrc

foamSystemCheck | tee log.foamSystemCheck

./Allwmake -with-bear -j 4 -s -q -l
alias of2512='source ~/openfoam/OpenFOAM-v2512/etc/bashrc'
